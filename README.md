# 命令行参数

``` bash
# 鸭皇存档API 的使用方法可以去主页的开发者文档去看
#
# 帮助
start.exe --help
start.exe -h
# 登录管理员
start.exe --admin {管理员密码}
# 退出管理员
start.exe --unadmin
# 管理员专属：
# 1. 调试终端
start.exe --shell
start.exe --console
# shell（调试终端）就是一个不完整的Python
# 所以有很多东西是不受支持的
# 例如导入一些库就是被受限制的
# 下面是可以被导入的库
# 1. requests
# 2. os
# 3. sys
# 4. pickleshare
# 5. code
#
# 以上的为本终端支持的包
# 其实支持的包少主要是要把游戏缩减，之前的300MB太大了
# 
# 2. 执行指令
start.exe --exec {command}
# command支持Python的能在lambda里执行的所有指令和鸭皇存档API
# 不支持使用的commands：（不是全部，"..."指数据）
# 1. import
# 2. from ... import ...
# 3. def ...():
# 4. for ... in ...:
# 5. if ...:
# 6. elif ...:
# 7. else:
# ...
# 其实就是import + 多行才能执行的语句
# 支持使用的commands：（不是全部，"..."指数据）
# 1. Setting("...")
# 2. [i for i in range(2)] # 单行的可以使用
# 3. 1 if x else 2 # tip: x未定义，这里只是一个例子，不是实际的代码
# 4. print("...") # 所有的python内置函数
# ...
#
# 设置一些参数
start.exe --config {config_name}={value}
# 参数：
# 1. fps -- 帧率（30 <= fps <= 120）
# 2. diamond -- 钻石（管理员专属）
# 3. coins -- 金币（管理员专属）
# 4. level -- 等级（管理员专属）
# 
```
